indicator("Passivation")


fmax(len) =>
    float max = 0
    for i=0 to len - 1
        max := if max < close[i]
            close[i]
        else
            max
    max
    
fmin(len) =>
    float min = close[0]
    for i=1 to len - 1
        min := if min > close[i]
            close[i]
        else
            min
    min

fdiff(len) => 
    fmax(len) - fmin(len)


passivation(close,p) => 
    float result = 0
    if close % p == 0
        result := close
    else
        result := close - close % p + p
    result

plot(passivation(close,2000),color=color.green)
plot(passivation(close,1000),color=color.red)
