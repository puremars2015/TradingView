indicator("Turtle Difference")


fmax(len) =>
    float max = 0
    for i=0 to len - 1
        max := if max < close[i]
            close[i]
        else
            max
    max
    
fmin(len) =>
    float min = close[0]
    for i=1 to len - 1
        min := if min > close[i]
            close[i]
        else
            min
    min

fdiff(len) => 
    fmax(len) - fmin(len)

plot(fdiff(40),color=color.white)
plot(fdiff(20),color=color.white)
